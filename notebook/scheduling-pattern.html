addresses = pd.read_csv('data/addresses.csv')
trips = pd.read_csv('data/trip_requests_v2.csv')
print("total # of addresses:" + str(len(addresses)))
print("total # of trip requests:" + str(len(trips)))

rides = trips.loc[trips['mode'] == 'passenger']
rides = rides.rename(columns={'id':'trip_id'})
rides = pd.merge(rides, addresses, left_on='from_address_id', right_on='id')
rides = pd.merge(rides, addresses, left_on='to_address_id', right_on='id', suffixes=('_orig', '_dest'))
print("# of passenger-mode trip requests:" + str(len(rides)))

# Join the addresses table with all driver-mode trip requests

# To find out what the most popular areas are for trip origins and destinations 
# passenger_orig_ct: how many passenger-mode trip origins are located in each zipcode zone
# passenger_dest_ct: how many passenger-mode trip destinations are located in each zipcode zone
# driver_orig_ct: how many driver-mode trip origins are located in each zipcode zone
# driver_dest_ct: how many driver-mode trip destinations are located in each zipcode zone

from point import *
from point_in_polygon import *

with open('data/zipcode.json') as f:
    zipcode_data = json.load(f)
df1 = pd.DataFrame(columns=['zipcode', 'passenger_orig_ct', 'passenger_dest_ct', 'driver_orig_ct', 'driver_dest_ct'])
i = 0
for feature in zipcode_data['features']:
    zipcode = feature['properties']['GEOID10']
    passenger_orig_ct = passenger_dest_ct = driver_orig_ct = driver_dest_ct = 0
    coords = feature['geometry']['coordinates']
    try:
        polygon = [[Point(p[0], p[1]) for p in points] for points in coords]
        for index, row in rides.iterrows():
            x0 = row['lng_orig']
            y0 = row['lat_orig']
            x1 = row['lng_dest']
            y1 = row['lat_dest']
            point0 = Point(x0,y0) # passenger-mode origin point
            point1 = Point(x1,y1) # passenger-mode destination point
            if pip_cross2(point0, polygon)[0] == True:
                passenger_orig_ct += 1
            if pip_cross2(point1, polygon)[0] == True:
                passenger_dest_ct += 1
        for index, row in drives.iterrows():
            x0 = row['lng_orig'] 
            y0 = row['lat_orig']
            x1 = row['lng_dest']
            y1 = row['lat_dest']
            point0 = Point(x0,y0) # driver-mode origin point
            point1 = Point(x1,y1) # driver-mode destination point
            if pip_cross2(point0, polygon)[0] == True:
                driver_orig_ct += 1
            if pip_cross2(point1, polygon)[0] == True:
                driver_dest_ct += 1
        df1.loc[i] = [zipcode, passenger_orig_ct, passenger_dest_ct, driver_orig_ct, driver_dest_ct]
        i += 1
    except:
        print("polygon error at " + zipcode)

# orig_ct: how many trip origins (both passenger- and driver-mode) are located in each zipcode zone
# dest_ct: how many trip destinations (both passenger- and driver-mode) are located in each zipcode zone
df1['orig_ct'] = df1['passenger_orig_ct'] + df1['driver_orig_ct']
df1['dest_ct'] = df1['passenger_dest_ct'] + df1['driver_dest_ct']
df1.to_csv('df1.csv', sep='\t', encoding='utf-8')

with open('data/zipcode.json') as f:
    zipcode_data = json.load(f)
my_features = []
features = zipcode_data['features']
for feature in features:
    geometry = feature['geometry']
    zipcode = feature['properties']['GEOID10']
    if len(df1.loc[df1['zipcode'] == int(zipcode)]) > 0:
        row = df1.loc[df1['zipcode'] == int(zipcode)].iloc[0]
        my_feature = {
            'geometry': geometry,
            'type': 'Feature',
            'properties': {
                'zipcode': zipcode,
                'orig_ct': int(row['orig_ct']),
                'dest_ct': int(row['dest_ct']),
                'passenger_orig_ct': int(row['passenger_orig_ct']),
                'passenger_dest_ct': int(row['passenger_dest_ct']),
                'driver_orig_ct': int(row['driver_orig_ct']),
                'driver_dest_ct': int(row['driver_dest_ct'])
            }
        }
        my_features.append(my_feature)
featureCollection = {
    'type': 'FeatureCollection',
    'features': my_features
}
with open('data/zipcode_areas.json', 'w') as outfile:
    json.dump(featureCollection, outfile)
